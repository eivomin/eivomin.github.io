## 5397.   키로거

---

### Code

```java
package Backjoon;

import java.io.*;
import java.util.*;

public class 키로거_5397 {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int N = Integer.parseInt(br.readLine());

        Stack<String> lstack;
        Stack<String> rstack;

        for(int i=0; i<N; i++){
            String str = br.readLine();
            int len = str.length();

            lstack = new Stack<>();
            rstack = new Stack<>();

            for(int j=0; j<len; j++){

                String tmp = String.valueOf(str.charAt(j));
                if(tmp.equals("<")){
                    if(!lstack.isEmpty()){
                        rstack.push(lstack.pop());
                    }
                }else if(tmp.equals(">")){
                    if(!rstack.isEmpty()){
                        lstack.push(rstack.pop());
                    }
                }else if(tmp.equals("-")){
                    if(!lstack.isEmpty()){
                        lstack.pop();
                    }
                }else
                    lstack.push(tmp);

                while(!lstack.isEmpty())
                    rstack.push(lstack.pop());

                StringBuilder sb = new StringBuilder();
                while(!rstack.isEmpty())
                    sb.append(rstack.pop());
                System.out.print(sb);
            }

        }

    }
}
```
