## 2630.   색종이 만들기

---

### Code

```java
package Backjoon;

import java.io.*;
import java.util.*;

public class 색종이만들기_2630 {
    static int N;
    static int[][] board;
    static int[] cnt;
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        N = Integer.parseInt(br.readLine());
        board = new int[N][N];
        cnt = new int[2];

        StringTokenizer st;

        for(int i=0; i<N; i++){
            st = new StringTokenizer(br.readLine(), " ");
            for(int j=0; j<N; j++){
                board[i][j] = Integer.parseInt(st.nextToken());
            }
        }

        papercnt(0, 0, N);

        System.out.println(cnt[0]+"\n"+cnt[1]);
    }

    static void papercnt(int x, int y, int size){
        if(isAll(x,y,size)){
            cnt[board[x][y]]++;
            return;
        }
        for (int i = 0; i < 2; i++) {
            for (int j = 0; j < 2; j++) {
                papercnt(x + (size / 2) * i, y + (size / 2) * j, size / 2);
            }
        }
    }

    static boolean isAll(int x, int y, int size){

        for(int i=x; i<x+size; i++){
            for(int j=y; j<y+size; j++){
                if(board[i][j] != board[x][y])
                    return false;
            }
        }
        return true;
    }
}
```
