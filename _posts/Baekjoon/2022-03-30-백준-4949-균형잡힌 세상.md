## 4949.   균형잡힌 세상

---

### Code

```java
package Backjoon;

import java.io.*;
import java.util.*;

public class 균형잡힌세상_4949 {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        while(true){
            String str = br.readLine();
            if(str.equals("."))
                break;
            int len = str.length();
            Stack<String> stack = new Stack<>();
            boolean flag = false;

            label: for(int i=0; i<len; i++){
                String s = str.charAt(i)+"";
                switch(s){
                    case "(":
                        stack.push(s);
                        break;
                    case ")":
                        if(stack.isEmpty()){
                            flag = true;
                            break label;
                        }
                        if(stack.peek().equals("(")){
                            stack.pop();
                            break;
                        }else{
                            flag = true;
                            break label;
                        }
                    case "[":
                        stack.push(s);
                        break;
                    case "]":
                        if(stack.isEmpty()){
                            flag = true;
                            break label;
                        }
                        if(stack.peek().equals("[")){
                            stack.pop();
                            break;
                        }else{
                            flag = true;
                            break label;
                        }
                }
            }

            if(stack.isEmpty() && flag == false)
                System.out.println("yes");
            else
                System.out.println("no");
        }


    }
}
```
