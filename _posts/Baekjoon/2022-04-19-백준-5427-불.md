## 5427.   불

---

### Code

```java
package Backjoon;

import java.io.*;
import java.util.*;

public class 불_5427 {
    static int w, h;
    static char[][] board;
    static int[][] dist1, dist2;
    static Queue<Pair> queue1, queue2;
    static int[] dx = {-1, 1, 0, 0};
    static int[] dy = {0, 0, -1, 1};
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int T = Integer.parseInt(br.readLine());

        StringTokenizer st;
        for(int tc=0; tc<T; tc++){
            st = new StringTokenizer(br.readLine(), " ");

            w = Integer.parseInt(st.nextToken());
            h = Integer.parseInt(st.nextToken());

            board = new char[h][w];
            dist1 = new int[h][w]; // 상근
            dist2 = new int[h][w]; // 불

            queue1 = new LinkedList<>(); // 상근
            queue2 = new LinkedList<>(); // 불

            for(int i=0; i<h; i++){
                String str = br.readLine();
                for(int j=0; j<w; j++){
                    char tmp = str.charAt(j);
                    board[i][j] = tmp;
                    if(tmp == '.'){
                        dist1[i][j] = -1;
                        dist2[i][j] = -1;
                    }else if(tmp == '@'){
                        dist1[i][j] = 0;
                        queue1.add(new Pair(i,j));
                    }else if(tmp == '*'){
                        dist2[i][j] = 0;
                        queue2.add(new Pair(i,j));
                    }
                }
            }

            while(!queue2.isEmpty()){
                Pair cur = queue2.poll();
                for(int dir=0; dir<4; dir++){
                    int nx = cur.x+dx[dir];
                    int ny = cur.y+dy[dir];
                    if(nx < 0 || nx >= h || ny < 0 || ny >= w) continue;
                    if(dist2[nx][ny] >= 0) continue;
                    if(board[nx][ny] == '.'){
                        dist2[nx][ny] = dist2[cur.x][cur.y]+1;
                        queue2.add(new Pair(nx,ny));
                    }
                }
            }

            boolean escape = false;
            while(!queue1.isEmpty() && (!escape)){
                Pair cur = queue1.poll();
                for(int dir=0; dir<4; dir++){
                    int nx = cur.x+dx[dir];
                    int ny = cur.y+dy[dir];
                    if(nx < 0 || nx >= h || ny < 0 || ny >= w) {
                        System.out.println(dist1[cur.x][cur.y]+1);
                        escape = true;
                        break;
                    }
                    if(dist1[nx][ny] >= 0) continue;
                    if(dist2[nx][ny] != -1 && dist2[nx][ny] <= dist1[cur.x][cur.y]+1) continue;
                    if(board[nx][ny] == '.'){
                        dist1[nx][ny] = dist1[cur.x][cur.y]+1;
                        queue1.add(new Pair(nx,ny));
                    }
                }
            }

            if(!escape)
                System.out.println("IMPOSSIBLE");

        }

    }

    static class Pair{
        int x, y;
        public Pair(int x, int y){
            this.x = x;
            this.y = y;
        }
    }
}
```
