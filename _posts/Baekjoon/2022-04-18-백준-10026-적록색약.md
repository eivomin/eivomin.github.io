## 10026.   적록색약

---

### Code

```java
package Backjoon;

import java.io.*;
import java.util.*;

public class 적록색약_10026 {

    static class Pair{
        int x, y;
        public Pair(int x, int y){
            this.x = x;
            this.y = y;
        }
    }
    static int N;
    static char[][] board1, board2;
    static boolean[][] visited;
    static Queue<Pair> queue;
    static int[] dx = {-1,1,0,0};
    static int[] dy = {0,0,-1,1};
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        N = Integer.parseInt(br.readLine());

        board1 = new char[N][N];
        board2 = new char[N][N];
        visited = new boolean[N][N];

        for(int i=0; i<N; i++){
            String str = br.readLine();
            for(int j=0; j<N; j++){
                char tmp = str.charAt(j);
                board1[i][j] = tmp;
                if(tmp == 'G')
                    board2[i][j] = 'R';
                else
                    board2[i][j] = tmp;
            }
        }

        int cnt1 = 0;
        for(int i=0; i<N; i++){
            for(int j=0; j<N; j++){
                if(!visited[i][j]){
                    bfs(i,j, board1);
                    cnt1++;
                }
            }
        }

        visited = new boolean[N][N];
        int cnt2 = 0;
        for(int i=0; i<N; i++){
            for(int j=0; j<N; j++){
                if(!visited[i][j]){
                    bfs(i,j, board2);
                    cnt2++;
                }
            }
        }

        System.out.println(cnt1+" "+cnt2);
    }

    static void bfs(int i, int j, char[][] board){
        queue = new LinkedList<>();

        char tmp = board[i][j];

        visited[i][j] = true;
        queue.add(new Pair(i,j));

        while(!queue.isEmpty()){
            Pair cur = queue.poll();
            for(int dir=0; dir<4; dir++){
                int nx = cur.x+dx[dir];
                int ny = cur.y+dy[dir];
                if(nx < 0 || nx >= N || ny < 0 || ny >= N) continue;
                if(visited[nx][ny] || board[nx][ny] != tmp) continue;
                visited[nx][ny] = true;
                queue.add(new Pair(nx,ny));
            }
        }
    }
}
```
